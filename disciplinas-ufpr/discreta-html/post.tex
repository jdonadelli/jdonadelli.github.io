\begin{definicao}
  Uma
  \begin{red}
    proposição
  \end{red}
  é uma sentença que pode assumir exatamente um de dois valores:
  \begin{red}
    VERDADEIRO
  \end{red}
   \emph{(V)} ou
   \begin{red}
     FALSO
   \end{red}
   \emph{(F)}. O valor é chamado de
  \begin{red}
    VALOR-VERDADE
  \end{red}
da proposição.
\end{definicao}

\begin{definicao}
  Seja $A$ uma proposição. A
  \begin{red}
    negação
  \end{red}
  de $A$ é a proposição $\nao(A)$ cujo
  valor-verdade é\\
    \begin{tabular}{cc}
      $A$ & $\nao(A)$\\
      $V$ & $F$\\
      $F$ & $V$\\
    \end{tabular}
\end{definicao}


\begin{definicao}
  Sejam $A$ e $B$ proposições, então a
  \begin{red}
    disjunção
  \end{red}
  é a proposição $A \ou B$
  cujo   valor-verdade é\\
    \begin{tabular}{ccc}
      $A$ & $B$ & $A \ou B$\\
      $V$ & $V$ & $V$ \\
      $V$ & $F$ & $V$ \\
      $F$ & $V$ & $V$ \\
      $F$ & $F$ & $F$ \\
    \end{tabular}
    a
    \begin{red}
      conjunção
    \end{red}
    é a proposição $A\ee B$ cujo valor-verdade é
    \begin{tabular}{ccc}
      $A$ & $B$ & $A\ee B$\\
      $V$ & $V$ & $V$ \\
      $V$ & $F$ & $F$ \\
      $F$ & $V$ & $F$ \\
      $F$ & $F$ & $F$ \\
    \end{tabular}
\end{definicao}

\begin{definicao}
  Sejam $A$ e $B$ proposições. 
  A
  \begin{red}
    condicional
  \end{red}
  é a proposição $A\Rightarrow B$ cujo valor-verdade é\\
    \begin{tabular}{ccc}
      $A$ & $B$ & $A \Rightarrow B$\\
      $V$ & $V$ & $V$ \\
      $V$ & $F$ & $F$ \\
      $F$ & $V$ & $V$ \\
      $F$ & $F$ & $V$ \\
    \end{tabular}
\end{definicao}

\begin{definicao}
  Sejam A e B proposições. 
  A
  \begin{red}
    bicondicional
  \end{red}
  é a proposição $A\Leftrightarrow B$ cujo valor-verdade é\\
    \begin{tabular}{ccc}
      $A$ & $B$ & $A \Leftrightarrow B$\\
      $V$ & $V$ & $V$ \\
      $V$ & $F$ & $F$ \\
      $F$ & $V$ & $F$ \\
      $F$ & $F$ & $V$ \\
    \end{tabular}
\end{definicao}


\begin{definicao}
  Duas proposições $A$ e $B$ são
  \begin{red}
    logicamente equivalentes
  \end{red}
  se assumem o mesmo valor-verdade. A notação é $A\equiv B$.
\end{definicao}



\textbf{Obs} $A\equiv B$ é o mesmo que dizer que $A\Leftrightarrow B$ é verdadeiro



\begin{teorema}\label{teo:1}
  $A \ou \nao (A)$.
\end{teorema}

\begin{teorema}[identidade]\label{teo:2}
  \begin{equation}
    A\ee V \equiv A
  \end{equation}
  \begin{equation}
    A\ou F \equiv A    
  \end{equation}
\end{teorema}

\begin{teorema}[dominação]\label{teo:3}
  \begin{equation}
    A\ou V \equiv V
  \end{equation}
  \begin{equation}
    A\ee F \equiv F    
  \end{equation}
\end{teorema}


\begin{teorema}[idempotência]\label{teo:4}
  \begin{equation}
    A\ee A \equiv A
  \end{equation}
  \begin{equation}
    A\ou A \equiv A    
  \end{equation}
\end{teorema}

\begin{teorema}[dupla negação]\label{teo:5}
  \begin{equation}
    \nao(\nao(A)) \equiv A
  \end{equation}
\end{teorema}

\begin{teorema}[distributiva]\label{teo:6}
  \begin{equation}
        A \ee (B \ou C) \equiv (A \ee B) \ou (A \ee C)
  \end{equation}
    \begin{equation}
          A \ou (B \ee C) \equiv (A \ou B) \ee (A \ou C)
    \end{equation}
\end{teorema}

\begin{teorema}[comutativa]\label{teo:7}
  \begin{equation}
    A \ou B  \equiv (B \ou A)
  \end{equation}
  \begin{equation}
    A \ee B  \equiv (B \ee A)
  \end{equation}
\end{teorema}

\begin{teorema}[associativa]\label{teo:8}
  \begin{equation}
    A \ou (B \ou C) \equiv (A \ou B) \ou C
  \end{equation}
  \begin{equation}
    A \ee (B \ee C) \equiv (A \ee B) \ee C
  \end{equation}
\end{teorema}

\begin{teorema}[De Morgan]\label{teo:9}
  \begin{equation}
    \nao(A \ou B)\equiv     \nao(A) \ee \nao(B)
  \end{equation}
  \begin{equation}
        \nao(A \ee B)\equiv     \nao(A) \ou \nao(B)
  \end{equation}
\end{teorema}

\begin{teorema}\label{teo:10}
  $
    A\Rightarrow B \equiv \nao(A) \ou B
  $.

Um
\begin{red}
  argumento
\end{red}
com premissas $P_1,P_2,\dots,P_n$ e conclusão $Q$ é
\begin{red}
  válido
\end{red}
quando
\begin{center}
  $Q$ é verdadeiro sempre que $P_1 \ee P_2 \ee \cdots \ee P_n$ é verdadeiro,
\end{center}
 ou seja
\begin{equation}
(  P_1 \ee P_2 \ee \cdots \ee P_n \Rightarrow Q ) \equiv V
\end{equation}

\begin{teorema}[Modus Ponens]\label{eq:modusponens}
  $ A \ee (A\Rightarrow B) \Rightarrow B $ é um argumento válido.
\end{teorema}


\begin{teorema}\label{teo:12}
  $ A\Rightarrow A \ou B $ é um argumento valido.
\end{teorema}


\begin{teorema}\label{teo:13}
  $ A \ee B \Rightarrow A $ é um argumento válido.
\end{teorema}

\begin{teorema}\label{teo:14}
  $ \nao(A\Rightarrow B) \equiv A \ee \nao(B) $.
\end{teorema}

\begin{teorema}\label{teo:vacuous}
  $F\Rightarrow A$.
\end{teorema}

\begin{teorema}\label{teo:16}
  $ \big( (A\Rightarrow B) \ee (A\Rightarrow C)\big) \Rightarrow
  (A\Rightarrow (B \ee C)) $ é um argumento válido.
\end{teorema}

\begin{teorema}\label{teo:17}
  $ (A\Rightarrow B) \ee (A\Rightarrow\nao(B)) \equiv (A=F)  $.
\end{teorema}

Uma
\begin{red}
  sentença aberta
\end{red}
é uma sentença parametrizada por uma ou mais variáveis. Uma sentença
aberta não tem valor lógico. O domínio da(s) variável(is) é chamado
\begin{red}
  universo do discurso
\end{red}.


O
\begin{red}
  conjunto verdade
\end{red}
de uma sentença aberta $P(x)$ sobre o universo $U$ é o conjunto
\begin{equation}
  \{ a\in U\colon P(a)\},
\end{equation}
ou seja, o conjunto formado pelos elementos de $U$ que tornam $P(x)$
verdadeira.

\textbf{Quantificador Universal} é a proposição
\begin{equation}
  \paratodo x,~P(x)
\end{equation}
significa que $P(x)$ é verdadeiro para todos os valores de $x$ no
universo do discurso.


\textbf{Quantificador Existencial} é a proposição
\begin{equation}
  \existe x,~P(x)
\end{equation}
significa que $P(x)$ é verdadeiro para algum valor de $x$ no universo
do discurso.


\begin{tabular}{ccc}
  proposição & verdadeira & falsa \\
  $\paratodo x,~P(x)$ & $P(x)=V$ para todo $x$ no universo do discurso&
  existe um $x$ no universo do discurso para o qual $P(x)$ é falso\\
  $\existe x,~P(x)$ & existe um $x$ no universo do discurso para o
  qual $P(x)$ é verdadeiro& $P(x)=F$ para todo $x$
\end{tabular}


Podemos caracterizar os quantificadores com proposições a cerca do
conjunto-verdade, 

\begin{definicao} Usaremos como definição as senguintes equivalências.
  
\begin{teorema}\label{teo:18}
  $
  \paratodo x,~P(x) \equiv \Big( \big\{x\in U\colon  P(x)\big\}=U\Big).
  $
\end{teorema}

\begin{teorema}\label{teo:19}
  $
  \existe x,~P(x) \equiv \Big( \big\{x\in U\colon  P(x)\big\}\neq \emptyset \Big).
  $
\end{teorema}
\end{definicao}


Com isso é fácil deduzir a negação das proposições quantificadas
\begin{teorema}\label{teo:20}
  $ \nao \Big( \paratodo x,~P(x)\Big) \equiv \existe x,~\nao(P(x)) $.
\end{teorema}

\begin{teorema}\label{teo:21}
  $ \nao\Big( \existe x,~P(x)\Big) \equiv \paratodo x,~\nao(P(x))$.
\end{teorema}


Em proposições com mais de uma variável
\begin{red}
  a ordem em que os quantificadores aparece é importante.
\end{red}
Por exemplo, se $x$ e $y$ são inteiros
\begin{equation}\label{eq:divide1}
  \paratodo x,~\existe y,~(x+y=0)
\end{equation}\begin{red}
  não é logicamente equivalente a 
\end{red}\begin{equation}\label{eq:divide2}
  \existe y,~\paratodo~x,~(x+y=0)
\end{equation}
pois \eqref{eq:divide1} é verdadeiro enquanto que \eqref{eq:divide2} é
falso.  Entretanto, em alguns casos vale a equivalência.  Por exemplo,
se $x$ e $y$ são números naturais
\begin{equation}
  \paratodo x,~\existe y,(x~\mathrm{divide}~y)
\end{equation}
é verdadeira, assim como
\begin{equation}
  \existe y,~\paratodo x,~(x~\mathrm{divide}~y)
\end{equation}
pois todo $x\in\N$ divide o $0$.


Se $X\subseteq U$ ou para explicitar o universo, escrevemos $\paratodo
x\in X,~P(x)$ que significa
\begin{equation}
  \paratodo x\in X,~P(x) \equiv \paratodo x,~(x\in X\Rightarrow P(x)).
\end{equation}



\begin{teorema}\label{teo:22}
  São argumento válidos:
  \begin{description}
  \item[instanciação universal:] Se $\paratodo x,~P(x)$ então $P(c)$
    sempre que $c\in U$;
  \item[generalização universal:] Se $P(c)$ para $c\in U$ arbitrário,
    então $\paratodo x,~P(x)$;
  \item[instanciação existencial:] Se $\existe x,~P(x)$ então $P(c)$
    para algum elemento $c\in U$;
  \item[generalização existencial:] Se $P(c)$ para algum $c\in U$
    específico, então $\existe x,~P(x)$.
  \end{description}
\end{teorema}

$A$ e $B$ agora denotam conjuntos.

$x\in A$ denota a proposição $x$ \textit{é elemento de} $A$.

$x\not\in A$ denota a proposição $\nao(x\in A)$.

$\emptyset$ denota o conjunto vazio.

$|A|$ denota a cardinalidade de $A$, isto é, o número de elementos quando $A$ é finito.


\begin{definicao}
  \begin{equation}\label{eq:contido}
    A\subseteq B \equiv \paratodo a(a\in A \Rightarrow a\in B).
  \end{equation}
  \begin{equation}
    A =  B \equiv     A\subseteq B \ee B\subseteq A.
  \end{equation}
  \begin{equation}
    x\in A \cup  B \equiv     (x\in A) \ou (x\in B).
  \end{equation}
  \begin{equation}
    x\in A \cap  B \equiv     (x\in A) \ee (x\in B).
  \end{equation}
  \begin{equation}
    x\in A - B \equiv     (x\in A) \ee (x\not\in B).
  \end{equation}
\end{definicao}

\begin{teorema}\label{teo:23}
  $A\subseteq A\cup B$.
\end{teorema}

\begin{teorema}\label{teo:24}
  $A\cap B \subseteq A$.
\end{teorema}

\begin{teorema}\label{teo:25}
  \begin{equation}
    A\cap (B\cap C) = (A\cap B)\cap C.
  \end{equation}
  \begin{equation}
    A\cup (B\cup C) = (A\cup B)\cup C    .
  \end{equation}
\end{teorema}

\begin{definicao}
  $A$ e $B$ são
  \begin{red}
    disjuntos
  \end{red}
  se $A\cap B=\emptyset$.
\end{definicao}


\begin{definicao}Sejam $A_1,A_2,\dots,A_n$ conjuntos e $n$ um natural
  positivo
  \begin{equation}
    x\in \bigcup_{i=1}^n A_i \equiv \existe i,~x\in A_i.
  \end{equation}
\end{definicao}

\begin{definicao}Sejam $A_1,A_2,\dots,A_n$ conjuntos e $n$ um natural positivo
  \begin{equation}
    x\in \bigcap_{i=1}^n A_i \equiv \paratodo i,~x\in A_i.
  \end{equation}
\end{definicao}


\begin{definicao}Sejam $A_1,A_2,\dots,A_n$ conjuntos e $n$ um natural positivo
  \begin{equation}
    \prod_{i=1}^n A_i = 
    \big\{ (a_1,a_2,\dots,a_n)\colon \paratodo i,~a_i\in A_i\big\}.
  \end{equation}
\end{definicao}

\textbf{Convenções}
Se $n=0$
\begin{equation}
  \bigcup_{i=1}^n A_i = \emptyset
  \quad \ee \quad
  \bigcap_{i=1}^n A_i = \emptyset.
\end{equation}
Se $A_i=A$ para todo $i$
\begin{equation}
  \prod_{i=1}^n A_i = A^n
\end{equation}



\begin{definicao}
$\{A_1,A_2,\dots,A_n\}$ é uma
\begin{red}
  partição
\end{red}
de $A$ se
\begin{enumerate}
\item $\paratodo i,~A_i\subseteq A$
\item $\paratodo i, \paratodo j,~(j\neq i\Rightarrow A_i\cap A_j =\emptyset)$
\item $\bigcup_{i=1}^n A_i = A$.
\end{enumerate}
\end{definicao}

\begin{exercicio}\label{exerc:1}
  Seja $C$ um conjunto. Prove o teorema: $\emptyset \subseteq C$.
\end{exercicio}
\begin{proof}
  Seja $C$ um conjunto. Vamos provar que $\emptyset \subseteq C$ é
  verdadeiro.  Pela definição (\ref{eq:contido}) $\emptyset \subseteq
  C$ é equivalente a $\paratodo a,~a\in \emptyset \Rightarrow a\in C$.

  $a\in \emptyset$ é falso e pelo teorema \ref{teo:vacuous}
  $F\Rightarrow A$ é uma proposição verdadeira, então $a\in \emptyset
  \Rightarrow a\in C$ é sempre uma proposição verdadeira, portanto
  também é $\emptyset \subseteq C$.
\end{proof}

\textbf{Notação} $[a..b]=\{z\in\Z\colon a\leq z\leq b\}$.

\textbf{Somatórios}

Seja $X\subseteq \R$ um
\href{http://en.wikipedia.org/wiki/Finite_set#Definition_and_terminology}{conjunto
  finito}. Denotamos por
\begin{equation}
  \sum_{a\in X} a
\end{equation}
a soma de todos os elementos de $X$. Como $X$ é um conjunto finito os
seus elementos podem ser enumerados
\begin{equation}
  a_1,a_2,\dots,a_n
\end{equation}
e também escrevemos
\begin{equation}
  \sum_{i=1}^n a_i \textrm{ para denotar } \sum_{a\in X} a.
\end{equation}

\begin{teorema}\label{teo:26}
  Sejam $c,a_1,a_2,\dots,a_n$ números reais e $X$  um conjunto
  finito. Prove
  \begin{enumerate}
  \item $\sum_{x\in X} c = c|X|$.
  \item $\sum_{i=1}^n(a_i+b_i) = 
    \sum_{i=1}^n a_i + \sum_{i=1}^n b_i $.
  \item $\sum_{i=1}^n c a_i = c \sum_{i=1}^n  a_i$. 
  \end{enumerate}
\end{teorema}


Um tipo particular de soma será bastante utilizado mais tarde, a soma
dos elementos de uma progressão aritmética e a soma dos elementos de
uma progressão geométrica. Seja
$$%\begin{equation}
  A = \{a_i\colon i\in [0..n-1]\}
$$%\end{equation}
onde $a_i\neq 0$ é um número real para todo $i\in[0..n-1]$.

$A$ é uma progressão aritmética de razão $r\in\R$ se
$$
a_i - a_{i-1} = r
$$
para todo $i\in [1..n-1]$.  

$A$ é uma progressão geométrica de razão $r\in\R$ se
\begin{equation}
  \label{eq:PG}
  {a_i} = r{a_{i-1}}
\end{equation}

para todo $i\in [1..n-1]$.  


\begin{teorema}\label{teo:27}
  Seja $A = \{a_i\colon i\in [0..n-1]\}$ uma progressão aritmética de
  razão $r$. Então
  \begin{equation}
    \sum_{a\in A} a = a_0n + \frac{n-1}2rn = \frac{(a_0 +a_{n-1})n}2.
  \end{equation}
\end{teorema}


\begin{teorema}\label{teo:28}
  Seja $A = \{a_i\colon i\in [0..n-1]\}$ uma progressão geométrica de
  razão $r\neq 1$. Então
  \begin{equation}
    \sum_{a\in A} a = a_0\frac{r^n-1}{r-1}.
  \end{equation}
\end{teorema}
\begin{proof}
  Seja $A = \{a_i\colon i\in [0..n-1]\}$ uma progressão geométrica de
  razão $r\neq 1$. Vamos provar que $\sum_{a\in A} a =
  a_0\frac{r^n-1}{r-1}$.

  Da equação \eqref{eq:PG} temos que $a_i = a_0 r^i$ portanto
  \[
    (r-1)\sum_{a\in A} a=
    (r-1)\sum_{i=0}^{n-1} a_i =
    (r-1)\sum_{i=0}^{n-1} a_0r^i 
  \]
  e usando o item 3 do teorema \ref{teo:26}
  \[
    (r-1)\sum_{i=0}^{n-1} a_0r^i =
    \sum_{i=0}^{n-1} (r-1)a_0r^i =
    \sum_{i=0}^{n-1} (a_0r^{i+1} - a_0r^i )
  \]
  e essa soma é
  \href{http://pt.wikipedia.org/wiki/Soma_telescópica}{telescópica},
  portanto 
  \[
    (r-1)\sum_{a\in A} a= a_0r^n - a_0
  \]
  fatorando o $a_0$ no lado direito e dividindo tudo por $r-1$ (o que
  podemos fazer pois $r\neq 1$), resulta em
   \[
    \sum_{a\in A} a= a_0\frac{r^n -1}{r-1}.
  \]
\end{proof}

Um
\begin{red}
  teorema
\end{red}
é uma sentença que pode ser demonstrado ser verdadeira. Demonstramos
que um teorema é verdadeiro exibindo uma sequência de proposições, que
chamamos de
\begin{red}
  prova
\end{red}
de modo que em cada passo usamos um argumento válido para ter um
conclusão verdadeira a partir de uma premissa verdadeira.



Vejamos algumas técnicas para construir provas de teoremas da forma
$A\Rightarrow B$.


\begin{red}
  \textbf{Prova direta}.
\end{red}
Pela definição de implicação $A=V$ é o único caso que precisamos
considerar para mostrar que a implicação $A\Rightarrow B$ é
verdadeira. Assumimos $A$ verdadeiro e usamos argumentos válidos para
concluir que $B$ é verdadeiro, usualmente, com várias concicionais
intermediarias e uma conclusão obtida por \textit{silogismo hipotético} (veja a
\href{http://www.inf.ufpr.br/jair/lista1.pdf}{lista 1}).

Por exemplo, se $n$ é um número natural então a \textit{implicação}
\begin{equation}
  \label{eq:ex1}
  3\mathrm{~divide~}n \Rightarrow
  9\mathrm{~divide~} n^2
\end{equation}
é verdadeira:
\begin{itemize}
\item Se $3\mathrm{~divide~}n $ é verdadeiro, então $n=3q$ é
  verdadeiro, por definição de \emph{divide};
\item se $n=3q$ é verdadeiro, então $n^2=(3q)^2$ é verdadeiro,
  elevando ao quadrado os dois lados da igualdade;
\item se $n^2=(3q)^2$ então $n^2 = 9q^2$;
\item se $n^2=9q^2$ então $9\mathrm{~divide~} n^2$ é verdadeiro, por
  definição de \textit{divide}.
\end{itemize}
Pelo silogismo hipotético: \textit{se $3\mathrm{~divide~}n $ então $
  9\mathrm{~divide~} n^2$}.


Como a variável $n$ acima pode assumir qualquer valor natural, ou
seja, $n$ é um elemento genérico de $\N$, o que provamos de fato foi,
pelo teorema \ref{teo:22} (generalização universal), que \[ \paratodo
n~( 3\mathrm{~divide~}n \Rightarrow 9\mathrm{~divide~} n^2).\]


Mais um exemplo:\\
\textbf{Teorema A}. \textit{Para todo $n\in\N$, se $n$ é ímpar, então
  $n^2$ é ímpar.}

\begin{proof}
  Seja $n$ um número natural (/*com isso podemos usar generalização
  universal*/).  Vamos provar que a implicação $n$ ímpar $\Rightarrow$
  $n^2$ ímpar é verdadeira.

  Suponha que $n$ é ímpar (/*se fosse par, a premissa da implicação
  seria falsa, portanto a implicação verdadeira*/).  Se $n$ é ímpar
  então $n=2k+1$ para algum $k\in \N$ pela definição de número ímpar.

  Se $n=2k+1$ então $n^2=(2k+1)^2$. Se $(2k+1)^2 = 2(2k^2+2k)+1$ então
  $n^2= 2(2k^2+2k)+1$, finalmente, se $n^2= 2(2k^2+2k)+1$ então $n^2$
  é ímpar.
\end{proof}


\begin{red}
  \textbf{Prova indireta}.  
\end{red}
Nesse tipo de prova, demonstramos que uma proposição logicamente
equivalente a $A\Rightarrow B$ é verdadeira.
\begin{description}
\item[ prova da contrapositiva: ] Provamos que \[\nao(B)\Rightarrow
  \nao(A)\] é verdadeira e concluímos que $A\Rightarrow B$ é
  verdadeira pela equivalência lógica
  \begin{equation}
    \label{eq:contrapositiva}
    A\Rightarrow B\equiv \nao(B)\Rightarrow   \nao(A).
  \end{equation}
  Por exemplo, a implicação
  \begin{equation}
    \label{eq:ex2}
    n^2 \mathrm{~par}
    \Rightarrow
    n \mathrm{~par}
  \end{equation}
  é verdadeira para todo número natural (tente uma prova direta). 

  Vamos provar a contrapositiva de \eqref{eq:ex2} \[n \mathrm{~impar}\Rightarrow n^2
  \mathrm{~impar}\]
  \begin{itemize}
  \item Suponha que $n$ é ímpar, então pelo \textbf{teorema A} acima
    $n^2$ é ímpar.
  \end{itemize}
  Por \eqref{eq:contrapositiva} temos \eqref{eq:ex2} verdadeira.

\item[ prova por vacuidade: ] A proposição $A\Rightarrow B$ é
  verdadeira porque conseguimos provar que $A$ é falso. Veja o
  exercício \ref{exerc:1} na aula 7.


\item[ prova por contradição: ] Na prova por contradição assumimos que
  a \textit{negação} da proposição a ser provada é verdadeira e
  derivamos disso uma
  \begin{red}
    contradição
  \end{red}, ou seja, uma proposição falsa.  De fato, se soubermos
  provar que $\nao(C)\Rightarrow F$ é verdadeiro, então concluiremos a
  partir de definição de condicional que $C$ é verdadeiro.

  No caso de provar que $A\Rightarrow B$ por contradição, provamos que
  $\nao(A\Rightarrow B) \Rightarrow F$ e verdadeiro, que equivale a $A
  \ee \nao(B) \Rightarrow F$ ser uma proposição verdadeira.

  O exercício \ref{exerc:2} mostra algumas equivalências usadas nessa
  técnica.  

  Vamos provar por contradição:\\
  \textbf{Teorema B}. \textit{Se $a$ e $b$ são números naturais
    coprimos, então não são ambos par.}\\

  \begin{proof}
    Vamos provar que
    \begin{eqnarray*}
      \paratodo a,~\paratodo b,~\big(
      \mathrm{mdc}(a,b)=1\Rightarrow \nao (a~\mathrm{par}\ee b~\mathrm{par}) \big)
    \end{eqnarray*}
    por contradição. 

    Sejam $a$ e $b$ números naturais, vamos provar que é verdade a
    proposição
    \begin{eqnarray*}
      \mathrm{mdc}(a,b)=1\ee 
      \nao\big(\nao (a~\mathrm{par}\ee b~\mathrm{par} ) \big)
      \Rightarrow 
      F.
    \end{eqnarray*}
    Suponha que $\mathrm{mdc}(a,b)=1\ee \big(a~\mathrm{par}\ee
    b~\mathrm{par} \big)$ é verdadeiro.

    Se $\mathrm{mdc}(a,b)=1\ee \big(a~\mathrm{par}\ee b~\mathrm{par}
    \big)$ então $\big(a~\mathrm{par}\ee b~\mathrm{par}
    \big)$.
    
    Se $(a~\mathrm{par}\ee b~\mathrm{par})$ então
    $\mathrm{mdc}(a,b)\geq 2$. Se $\mathrm{mdc}(a,b)\geq 2$ então
    $\mathrm{mdc}(a,b)\neq 1$ portanto, por silogismos hipotético \[
    \mathrm{mdc}(a,b)=1\ee \big(a~\mathrm{par}\ee b~\mathrm{par} \big)
    \Rightarrow \mathrm{mdc(a,b)\neq 1}\] cuja conclusão é uma contradição,
    ou seja, $\mathrm{mdc(a,b)\neq 1}$ é falso.
  \end{proof}

Vamos provar por contradição:\\
\textbf{Teorema C}. $\sqrt{2}$ é irracional.

\begin{proof}
  Vamos provar que \[ \sqrt{2} \not\in \Q \] por contradição, ou seja, vamos
  provar que \[ \sqrt{2} \in \Q \Rightarrow F.\]
  Suponha que $\sqrt{2} \in \Q $.

  Se $\sqrt{2} \in \Q $ então, por definição de $\Q$, existem naturais
  positivos $d$ e $e$ tais que $\sqrt{2}=\frac{d}{e}$.  

  Façamos $$a=\frac d{\mathrm{mdc(d,e)}} \textrm{~e~}b=\frac
  e{\mathrm{mdc(d,e)}}$$ e temos $\mathrm{mdc}(a,b)=1$ pelo teorema
  B', portanto, até aqui provamos \[ \sqrt{2} \in \Q \Rightarrow
  \existe a,~\existe b, \left( \sqrt{2}=\frac{a}{b}\ee
    \mathrm{mdc}(a,b)=1\right).\]

  Se $\sqrt{2} =\frac ab$ então $2 = a^2 / b^2$. \\
  Se $2 = a^2 / b^2$ então $a^2=2 b^2$.\\
  Se $a^2=2 b^2$ então $a^2$ é par.  \\
  Se $a^2$ é par então $a$ é par, pelo teorema A acima.\\
  Se $a$ é par então existe $k\in\N$, $a=2k$.\\
  Se $a=2k$ então $2b^2 = 4k^2$.\\
  Se $2b^2 = 4k^2$ então $b^2 = 2k^2$.\\
  Se $b^2 = 2k^2$ então $b^2$ é par. \\
  Se $b^2$ então $b$ é par. 

  Se $a$ é par e $b$ é par, então $\mathrm{mdc}(a,b)\geq 2$, que é uma
  contradição.
\end{proof}
\end{description}

  \item $(P\Rightarrow Q) \equiv ((P\ee \nao Q)\Rightarrow \nao P)$.

\end{exercicio}

\begin{proof}[ \textit{Outra prova do teorema C} ]
  Seja $x=\sqrt{2}$. Vamos provar que \[ x=\sqrt{2}\Rightarrow
  x\not\in\Q\] por contradição. De fato, provaremos que existem
  naturais positivos $a$ e $b$ tais que
  \begin{equation}
    \label{eq:q3}
    x=\sqrt{2} \ee  x\in\Q \Rightarrow x=\frac{a}{b}\ee \mathrm{mdc}(a,b)=1 \\
  \end{equation}  \begin{equation}
    \label{eq:q4}
    x=\sqrt{2} \ee  x\in\Q \Rightarrow x=\frac{a}{b}\ee \mathrm{mdc}(a,b)\neq 1
  \end{equation}
  e derivaremos daí uma contradição.

  Vamos provar \eqref{eq:q3}.  Suponha que $x\in \Q$. Então existem
  naturais positivos $d$ e $e$ tais que $x=\frac{d}{e}$. Tomemos
  $a=\frac d{\mathrm{mdc(d,e)}}$ e $b=\frac e{\mathrm{mdc(d,e)}}$,
  logo $\mathrm{mdc}(a,b)=1$.

  Agora, vamos provar \eqref{eq:q4}.  Se $x =\frac ab$ então $x^2 =
  a^2 / b^2$, ou seja, $2 = a^2 / b^2$. Multiplicando ambos os lados
  por $b^2$ temos $2b^2 = a^2$, logo $a^2$ é par.  Se $a^2$ é par então
  $a$ é par, pelo teorema A acima. Se $a$ é par então $a=2k$ e
  substituindo em $2b^2 = a^2$ deduzimos $2b^2 = 4k^2$, logo $b^2 =
  2k^2$, ou seja $b^2$ é par. Pelo mesmo argumento dado anteriormente,
  $b$ é par. Portanto, $\mathrm{mdc}(a,b)\geq 2$.

  Usamos o teorema \ref{teo:16} e a partir de \eqref{eq:q3} e
  \eqref{eq:q4} obtemos
  \begin{equation}\label{eq:q1}
    x=\sqrt{2} \ee  x\in\Q \Rightarrow 
    \Big(x=\frac{a}{b}\ee \mathrm{mdc}(a,b)=1 \ee \mathrm{mdc}(a,b)\neq 1\Big) 
  \end{equation} e simplificando o lado direito (teorema \ref{teo:13})
  \[
    \Big(x=\frac{a}{b}\ee \mathrm{mdc}(a,b)=1 \ee \mathrm{mdc}(a,b)\neq 1 \Big)
    \Rightarrow
  \]  \begin{equation}\label{eq:q2}
    \Big(\mathrm{mdc}(a,b)=1 \ee \mathrm{mdc}(a,b)\neq 1 \Big)
  \end{equation}
  finalmente, pelo \textit{silogismo hipotético} (veja (16) na lista
  1) de \eqref{eq:q1} e \eqref{eq:q2}
  \begin{eqnarray*}
    x=\sqrt{2} \ee  x\in\Q \Rightarrow  
    \Big(\mathrm{mdc}(a,b)=1 \ee \mathrm{mdc}(a,b)\neq 1 \Big)
  \end{eqnarray*}
  e a conclusão é uma contradição.
\end{proof}


\begin{proof}[ \textit{Outra prova do teorema B} ]
    Sejam    $a$ e $b$ números naturais. Vamos provar que
    \begin{eqnarray*}
      \mathrm{mdc}(a,b)=1\Rightarrow 
      \nao\big(a~\mathrm{par}\ee b~\mathrm{par} \big).
    \end{eqnarray*}
    Por \eqref{eq:contrapositiva} basta provarmos a contrapositiva
    \begin{eqnarray*}
      \big(a~\mathrm{par}\ee b~\mathrm{par} \big) 
      \Rightarrow \nao(\mathrm{mdc}(a,b)=1).
    \end{eqnarray*}
    
    Suponha $a$ par e $b$ par. Então $2$ divide $a$ e $2$ divide $b$,
    por definição de \textit{par}, portanto, $\mathrm{mdc}(a,b)\geq
    2$, ou seja, $\nao(\mathrm{mdc}(a,b)=1)$ é verdadeiro.
  \end{proof}

\textbf{Demonstrações} de proposições quantificadas existencialmente:
Para demonstrar uma proposição da forma $\existe x,~P(x)$, usualmente,
ou
\begin{itemize}
\item
  \begin{red}
    (prova construtiva)
  \end{red}
exibimos um elemento $c$ do universo tal que $P(c)$ seja verdade, ou
\item
  \begin{red}
    (prova não-construtiva)
  \end{red}
  ou inferimos indiretamente a existência desse objeto que torna $P$
  verdadeira, por exemplo, através de uma prova por contradição:
  assumimos que tal objeto não existe e derivamos uma contradição.
\end{itemize}


Exemplos:


\textbf{Teorema D}. Existe um inteiro positivo que pode ser escrito como
a soma de dois cubos de duas maneiras diferentes.

\begin{proof}[Prova construtiva]
 É suficiente verificar que $1729 = 10^3 + 9^3 = 12^3 + 1^3$.
\end{proof}


\textbf{Teorema E}. Existem $x$, $y$ irracionais tais que $x^y\in\Q$.

\begin{proof}[Prova não-construtiva]
  Vimos na aula anterior que $\sqrt{2}$ é irracional. Se
  $\sqrt{2}^{\sqrt{2}}$ é racional então tomamos $x=y=\sqrt 2$, senão
  tomamos $x=\sqrt{2}^{\sqrt{2}}$ e $y=\sqrt{2}$ e temos $x^y =
  {\sqrt{2}^{\sqrt{2}}}^{\sqrt{2}} = \sqrt{2}^2 =2$ que é racional.
\end{proof}


\textbf{Teorema F}. Existe uma partição $\{A,B\}$ de $\Z$ com $A$ e $B$ infinitos.

\begin{proof}[Prova construtiva]
  Defina $A = \{2k \colon k\in\Z\}$ e $B = \{2k+1 \colon
  k\in\Z\}$. Nenhum desses conjuntos é finito (dê uma prova por
  contradição) e definem  uma partição dos inteiros (prove).
\end{proof}


\textbf{Teorema G}. Se $\frac pq \in \Q$ então existe $n\in\Z$, $\frac pq < n$.

\begin{proof}[Prova construtiva]
  Se $p=0$ então tomamos $n=1$, senão tomamos $n=|p|+1$. No segundo
  caso $\frac pq \leq |\frac pq | <  |p|+1$ pois $|q|\geq 1$.
\end{proof}

\textbf{Teorema H}. O poninômio $p(x)=x^3+x-1$ tem exatamente uma raiz real.

\begin{proof}[Prova não-construtiva, precisa de cáculo 1]
  Pelo
  \href{http://pt.wikipedia.org/wiki/Teorema_do_Valor_Intermediário}{Teorema
    do Valor Intermediário} para todo $b\in [p(0),p(1)]$, existe $a\in
  [0,1]$ tal que $p(a)=b$.

  Mas $p(0)=-1$ e $p(1)=1$ então podemos tomar $b=0$ e concluir que existe $a\in
  [0,1]$ tal que $p(a)=0$.
  
  Resta provar que a raiz é única. (Exercício. Dica: use o 
  \href{http://pt.wikipedia.org/wiki/Teorema_do_valor_médio}{Teorema
    do Valor Médio} e prove por contradição.)
\end{proof}

\textbf{questão 3 - 10 pts} Prove por contradição: $(A-B) \cap B=\emptyset$.

 
\fbox{

    Vamos provar  $(A-B) \cap B\neq \emptyset\Rightarrow F$.\\

    Se $(A-B) \cap B\neq \emptyset$ então existe $x$, $x\in (A-B) \cap B$.\\

    Portanto, $(A-B) \cap B\neq \emptyset\Rightarrow F$.

      \textbf{Teorema} \emph{Sejam $a$, $b$, $c$, $d$ números naturais.  Se
      $c$ divide $a$ e $c$ divide $b$ e $d$ divide $a$ e $d$ divide $b$
      e $c$ não divide $d$, então $dc$ divide $a$ e $dc$ divide $b$.}
 \hrule 

      \begin{proof}
        Se $d$ divide $a$ e $b$, então existem $x$ e $y$ tais que $a=xd$
        e $b=yd$.
        
        Se $c$ divide $a$ então $c$ divide $xd$.\\
        Se $c$ divide $xd$ e não divide $d$, então $c$ divide $x$.
        
        Analogamente, se $c$ divide $b$ então $c$ divide $yd$.\\
        Se $c$ divide $yd$ e não divide $d$, então $c$ divide $y$.
        
        Se $c$ divide $x$ e $y$, então existem $z$ e $w$ tais que $x=cz$
        e $y=cw$.
        
        Das conclusões acima temos $a=xd = (cz)d = (dc)z$ e $b= yd =
        (cw)d = (dc)w$, portanto, $dc$ divide $a$ e $dc$ divide $b$.      
      \end{proof}
 \hrule

derivável em $(a,b)$ então existe $c\in (a,b)$ tal que

A prova de invariantes ficará pra depois de indução.

Um programa está
\begin{red}
  correto
\end{red}
se a saída está correta para toda entrada. A
prova da \emph{correção do programa} tem duas partes: provar que a
resposta está correta se o programa terminar (
\begin{red}
  correção parcial
\end{red}
), provar que o programa termina.



A seguir, $p$ e $q$ são proposições.

\begin{definicao}
  Um programa, ou trecho de programa, $S$ está
  \begin{red}
    parcialmente correto
  \end{red}
  com respeito a \textbf{pré-condição} $p$ e a \textbf{pós-condição}
  $q$ se sempre que $p$ for verdadeiro para os valores de entrada de
  $S$ e $S$ terminar, então $q$ é verdadeiro para os valores de saída de $S$.
\end{definicao}


\textbf{Notação} $p\{S\}q$ significa \emph{$S$ está parcialmente
  correto com respeito a pré-condição $p$ e a pós-condição $q$}.


\textbf{Exemplo} Por exemplo, se $p$ é $x=1$ e $q$ é $z=3$ e $S$ é o
trecho de programa 
\begin{verbatim}
    y:=2; 
    z:= x + y;
\end{verbatim}

então, $p\{S\}q$.  De fato, suponha $p$. Após a execução de $S$, $y=2$
e $z=1+2$ pois de $p$ temos $x=1$. Portanto, $z=3$.

\textbf{Argumentos válidos} para correção
\begin{red}
  parcial
\end{red}
de programas:

1.~\textbf{Composição}
\begin{eqnarray*}
  p\{S_1\}q \ee q\{S_2\}r \Rightarrow p\{S_1;S_2\}r
\end{eqnarray*}

2.~\textbf{Condicional}

2.1.~No caso 

\begin{verbatim}
    Se (condição) então S;
\end{verbatim}


o argumento é
\begin{eqnarray*}
&  \big(  ( p\ee \textrm{condi\c c\~ao}) \{S\}q \ee 
  ( p\ee \nao(\textrm{condi\c c\~ao})) \Rightarrow 
  q \big)& \\
  &\Longrightarrow& \\
  &p\{ \texttt{Se (condi\c c\~ao) ent\~ao S}\}q&
\end{eqnarray*}

2.2.~No caso 

\begin{verbatim}
    Se (condição) então S_1;
       senão S_2;
\end{verbatim}


o argumento é
\begin{eqnarray*}
  &\big(  ( p\ee \textrm{condi\c c\~ao}) \{S_1\}q \ee 
  ( p\ee \nao(\textrm{condi\c c\~ao})) \{S_2\}q \big)&\\
  &\Longrightarrow&\\
&  p\{ \texttt{Se (condi\c c\~ao) ent\~ao }S_1;\texttt{ sen\~ao }S_2\}q&
\end{eqnarray*}


3.~\textbf{Laços}

No caso

\begin{verbatim}
    Enquanto (condição) 
           S;
\end{verbatim}

o argumento é
\begin{eqnarray*}
  &( p\ee \textrm{condi\c c\~ao}) \{S\}p &\\
  &  \Longrightarrow&\\
  &p\{ \texttt{Enquanto (condi\c c\~ao) S}\}(\nao(\textrm{condi\c c\~ao})\ee p)&
\end{eqnarray*}

A proposição $(p\ee \textrm{condi\c c\~ao}) \{S\}p $ é o
\begin{red}
  invariante do laço
\end{red}
que por enquanto vamos assumir que é dado e é verdadeiro. Provar que
uma proposição é invariante de laço precisa de indução, que veremos mais tarde.

Um dos \href{http://mathworld.wolfram.com/PeanosAxioms.html}{axiomas
  de Peano} para os números naturais diz que para $X\subseteq \N$

\begin{equation}
  \label{eq:peano}
  0\in X \ee \big( \paratodo n,~(n\in X\Rightarrow n+1\in X)\big) 
\Longrightarrow
X=\N.
\end{equation}


Com isso temos o seguintes argumentos válidos: seja $P(n)$ uma sentença aberta.

\begin{teorema}\label{teo:PIF1}
  Se \[P(0)\ee \paratodo n\in \N,~\big(P(n) \Rightarrow P(n+1)\big)\]
  então \[\paratodo n\in \N,~P(n).\]
\end{teorema}

Provaremos:

\begin{teorema}\label{teo:PIF2} Seja $a\in\N$.
  Se \[P(a)\ee \paratodo n\geq a,~\big(P(n) \Rightarrow P(n+1)\big)\]
  então \[\paratodo n\geq a,~P(n).\]
\end{teorema}
\begin{proof}
  Para
  \[
  X=\{m \in\N \colon P(m+a)\}.
  \]
  se $P(a)$ então $0\in X$. Seja $n\geq a$ e defina $m=n-a$. Se $P(n)
  \Rightarrow P(n+1)$ então $m\in X \Rightarrow m+1 \in X$. Por
  \eqref{eq:peano} $X=\N$, portanto, $P(n)$ para todo $n\geq a$.
\end{proof}


Indução matemática é uma técnica de prova para proposições da forma
$\paratodo n\geq a,~P(n)$. Numa prova por indução provamos $P(a)$ que é
dito a \textbf{base} da indução, e provamos $\paratodo n\geq a,~\big(P(n)
\Rightarrow P(n+1)\big)$ que é dito o \textbf{passo} da indução. 

\textbf{Exemplo} Para todo $n\in\N$, $\sum_{i=0}^n i = n(n+1)/2$.

\textbf{Exemplo} Para todo $n \geq  5$, $2^n > n^2$.

\textbf{A base é importante}, sem ela poderíamos pensar que sabemos
provar
\begin{center}
  $n(n+1)$ é ímpar para todo $n\geq1$  
\end{center}
que obviamente não vale, pois conseguimos provar a implicação do passo
da indução para essa sentença. Vamos provar que
\begin{center}
  para todo $n\geq1$, $n(n+1)$ ímpar $\Rightarrow (n+1)(n+2)$ ímpar.
\end{center}
Seja $n>1$ um natural e suponha que $n(n+1)$ é ímpar. Então
\begin{eqnarray*}
  (t+1)(t+2) = (t+1)t + (t+1)2
\end{eqnarray*}
que é da forma ímpar $+$ par, portanto ímpar.


\textbf{A passo é importante}, uma prova descuidada pode por tudo a
perder. Por exemplo, seja $P(t)$ a sentença
\[
\paratodo a\in\N,~\paratodo b\in\N, (\max\{a,b\}=t \Rightarrow a=b).
\]

Se $\max\{a,b\}=0$ então $a=b=0$, portanto a base é verdadeira.

Seja $t\in\N$,  suponha que 
\[
\paratodo a,~\paratodo b, (\max\{a,b\}=t-1 \Rightarrow a=b)
\]
e vamos provar que $\paratodo a,$ $\paratodo b,$ $(\max\{a,b\}=t
\Rightarrow a=b)$. Se $\max\{a,b\}=t$ então $\max\{a-1,b-1\}=t-1$,
portanto pela hipótese acima $a-1=b-1$, logo $a=b$.

Claramente, $P(t)$ é falso (determine um contraexemplo). O problema
com o passo é que $P(0) \not\Rightarrow P(1)$ quando $a$ ou $b$ vale
$0$.

 \textbf{Boa-Ordenação}

Seja $A\subset \R$. O
\begin{red}
  menor elemento
\end{red} 
de $A$, quando existe, é o elemento $m \in A$ tal que $\paratodo a\in
A$, $m \leq  a$.

\textbf{Notação.} $\min A$ denota o menor elemento de $A$, se ele existe.

\textbf{Exemplo.} $A=(0,1)\subset \R$ não tem menor elemento. (prove)

\textbf{Exemplo.} $A=[0,1)\subset \R$ tem menor elemento, $\min A =0$. (prove)


\begin{teorema}
  \label{teo:min}
  Se $A\subset \N$ e 
  \[\big( \paratodo x\in [0..n], ~ x\not\in A \big)
  \ee \big(\existe y\in [0..n+1], ~y\in A\big).\]
  então $n+1$ é o menor elemento de $A$.
\end{teorema}
\begin{proof}
  Se $\paratodo x\in [0..n], ~ x\not\in A \ee \existe x\in [0..n+1],
  ~x\in A$ então $n+1 \in A$.

  Agora, precisamos provar que $\paratodo a\in A$, $n+1 \leq  a$.

  Por contradição. Suponha que $\nao(\paratodo a\in A$, $n+1 \leq  a)$,
  ou seja, \[\existe a\in A,~ a < n+1.\] 

  Seja $c\in A$ tal que $c < n+1$ (instanciação existencial, teorema
  \ref{teo:22}). Se $c < n+1$ então $c\in [0..n]$ portanto $c\not\in
  A$, portanto
  \[\existe a\in A,~ a < n+1 \Rightarrow c\in A \ee c\not\in   A \]
  essa última proposição é uma contradição.
\end{proof}

\begin{teorema}
  \label{teo:BO}
    Seja $A\subset \N$. Se $A\neq \emptyset$ então $A$ tem um menor elemento.
\end{teorema}
\begin{proof}
  Seja $A$ um subconjunto não-vazio dos naturais. Vamos provar que $A$
  tem menor elemento em dois casos: $0\in A$ e $0\not\in A$.

  No caso $0\in A$, o $0$ é o menor elemento de $A$.

  Assumimos que $0 \not\in A$ e definimos
  \[
  X= \{ n\in \N \colon [0..n] \subset \N - A \}.
  \]
  $0\in X$, pois se $0 \not\in A$ então $0 \in \N - A$, logo $\{0\}
  \subset \N - A$. Notemos que $\{0\} = [0..0]$.

  Se $A\neq \emptyset$ então $X\neq \N$. 

  Se $0\in X$ e $X\neq \N$ então pela a contrapositiva do axioma de
  Peano \eqref{eq:peano}
  \[
  X\neq \N
  \Longrightarrow
  0\not\in X \ou \nao \big( \paratodo n,~(n\in X\Rightarrow n+1\in X)\big) 
  \]
  concluímos que 
  \[
  \nao \big( \paratodo n,~(n\in X\Rightarrow n+1\in X)\big)
  \]
%
  \begin{eqnarray*}
%  \nao \big( \paratodo n,~(n\in X\Rightarrow n+1\in X)\big)
    & \equiv & \existe n, \nao \big( n\in X\Rightarrow n+1\in X \big)\\
    & \equiv & \existe n, \nao \big( \nao(n\in X)\ou  (n+1\in X) \big)\\
    & \equiv & \existe n, \big( (n\in X)\ee \nao(n+1\in X) \big)
  \end{eqnarray*}

  Seja $m$ tal que $m\in X$ e $m+1\not\in X$.\\
  Agora, usando a definição de $X$, \\
  se $m\in X$ então $[0..m]\subset \N -A$.\\
  $[0..m]\subset \N -A \equiv \paratodo x\in [0..m], ~ x\not\in A $;\\
  se $m+1\not\in X$ então $[0..m+1]\not\subset \N -A$;\\
  $[0..m+1]\not\subset \N -A \equiv \existe y\in [0..m+1],~y\in A$;\\
  portanto
  \[
  ( \paratodo x\in [0..m], ~ x\not\in A )\ee
  ( \existe y\in [0..m+1],~y\in A )
  \]
  e pelo teorema \ref{teo:min}, $m+1$ é o menor elemento de $A$.
\end{proof}


\textbf{Exemplo.} para todo inteiro $n\geq 5$, existem naturais $a$ e

\begin{proof}

\end{proof}

\textbf{Invariante de laço}.

Consideremos o seguinte algoritmo:

\begin{verbatim}
fat(n)
/* Dados  : n natural */
/* Devolve: n !*/

   i:=1;
   f:=1; 
   enquanto (i < n)
    {
      f := f*(i+1);
      i := i + 1;
    }
   devolva f.
\end{verbatim}


Vamos provar que a seguinte proposição é um invariante do laço na
terceira linha do algoritmo
\begin{equation}
  \label{eq:invariante1}
  i \geq 1 \ee i \leq n \ee f = i! 
\end{equation}
Antes, porém, notemos que o invariante e a negação da condição do laço
resulta que quando o laço termina temos $i \geq 1$ e $i \leq n$ e $f =
i!$ e $\nao(i<n)$, ou seja, $i=n$ e $f = n!$, provando a correção
parcial do algoritmo.

Vamos provar que a proposição \eqref{eq:invariante1} é verdadeira
independente do número de vezes que o laço é executado. A prova é por
indução no número de iterações do $\texttt{enquanto}$.

Suponhamos que a condição do laço vale, ou seja, $i<n$. Antes da
primeira iteração \eqref{eq:invariante1} é verdadeiro pois $i=f=1!=1$.

Suponhamos que $i<n$ e que $ i \geq 1 \ee i \leq n \ee f = i! $. Após
a execução de $\texttt{\{ f := f*(i+1);}$ e após a execução de
$\texttt{\{ i := i+1;\}}$ temos que para o novo valor de $i$, $i\geq  1$ e
$i \leq n$ e $f = i!$.


Voltemos ao exemplo da {aula 12}.

\begin{verbatim}
/* m e n inteiros e a=|n| e k=0 e x=0 */

   enquanto (k<a)

       /* invariante:  x=mk e k <= a */

     {  x:=x+m;
        k:=k+1; }
\end{verbatim}

Suponha que a condição do laço, $k<a$, é verdadeira.  Antes da
primeira iteração o invariante é verdadeiro.

Suponha que a condição do laço, $k<a$, é verdadeira e que vale o
invariante. Considere uma execução de $\texttt{\{ x:=x+m; k:=k+1; \}}$ e
denote por $x'$ e $k'$ os valores das variáveis $x$ e $k$ \emph{antes
  dessa execução}.  Então, por hipótese \[x'=mk' \ee k'\leq a\] e
(condição) $k'<a$; após a execução temos $x=x'+m$ e $k=k'+1$.

Se $x=x'+m$ e $x'=mk'$, então $x=(mk')+m = m(k'+1)$. \\
Se $k=k'+1$ $x= m(k'+1)$ então $x=mk$. \\
Se $k' < a $ então $k'+1 \leq a$.\\
Portanto, após a execução o invariante é verdadeiro


\textbf{Algoritmos recursivos}.

\textbf{Exemplo 1}.

\begin{verbatim}
fat(n)

   /* Dados  : n natural */
   /* Devolve: n !*/

1.   se (n=0) devolva 1;
2.   devolva n*fat(n-1).
\end{verbatim}


A prova da correção é fácil usando indução.

\begin{proof}
  Vamos provar que fat$(n)=n!$ usando indução em $n\in\N$. Se $n=0$
  então o algoritmo devolve $1$ na linha 1, ou seja, então fat$(n) =1$

  Seja $n \in \N$. Assuma que fat$(n)=n!$ e vamos provar que
  fat$(n+1)=(n+1)!$.  Como $n+1\geq 1$, pelo algoritmo fat$(n+1)=
  (n+1)*$fat$(n)$ e por hipótese fat$(n)=n!$, portanto, fat$(n+1)=
  (n+1)*n! = (n+1)!$. Pelo teorema \ref{teo:PIF1} temos fat$(n)=n!$
  para todo $n\in \N$.
\end{proof}


\textbf{Exemplo 2}.

\begin{verbatim}
pot(a,n)

   /* Dados  : a real positivo e n natural */
   /* Devolve: a^n */

1.   se (n=0) devolva 1;
2.   devolva a*pot(a,n-1).
\end{verbatim}


\begin{proof}

  Seja $n\in\N$ um natural arbitrário e suponha que pot$(a,n)=a^n$.
  Então $n+1\geq 1$ portanto, pelo algoritmo, pot$(n+1) = a*$pot$(a,n)
  = a*a^n = a^{n+1}$.
\end{proof}

\textbf{Função teto} $\lceil \; \rceil \colon \R \to \Z$ dada por \[

\textbf{Função chão} $\lfloor \; \rfloor \colon \R \to \Z$ dada por \[

\begin{teorema}\label{teo:teto}
  Sejam $x \in \R$ e $t\in \Z$.\\
  $\lceil \; \rceil \colon \R \to \Z$ satisfaz
  \begin{enumerate}
  \item $\lceil x \rceil = x \Leftrightarrow x\in \Z$.
  \item $\lceil x \rceil = t \Leftrightarrow t-1 < x \leq t$,
  \item $\lceil x \rceil = t \Leftrightarrow x \leq t < x+1$,
  \item $\lceil \; \rceil$ é não-decrescente em $\R$,
  \item $\lceil x+t \rceil = \lceil x \rceil + t$.
  \end{enumerate}
\end{teorema}
\begin{proof}
  \begin{itemize}%$\lceil \rceil$
  \item Vamos provar 1. Primeiro suponhamos que $\lceil x \rceil =
    x$. Então $\lceil x \rceil \in \Z$ por definição de $\lceil \;
    \rceil$. Agora, suponhamos que $x \in \Z$. Então $x \in \{ z\in \Z
    \colon z\geq x \}$ e como $\lceil x \rceil = \min \{ z\in \Z
    \colon z\geq x \}$ temos
    \[
    x \geq \lceil x \rceil.
    \]
    Por outro lado, $\lceil x \rceil \in \{ z\in \Z \colon z\geq x \}$
    por definição de \emph{menor elemento} (aula 15), portanto
    \begin{equation}
            \label{eq:teto}
      x \leq \lceil x \rceil
    \end{equation}
    donde concluímos que $\lceil x \rceil = x$.

  \item Vamos provar 2. Suponhamos que $\lceil x \rceil = t$. Se
    $\lceil x \rceil = t$ então $t-1 < x$ pois, caso contrário, isto é
    se $t-1 \geq x$ então $t -1 \in \{ z\in \Z \colon z\geq x \}$ e $t
    -1 <t $ em que $t = \min \{ z\in \Z \colon z\geq x \}$, uma
    contradição. Por \eqref{eq:teto} $x \leq t$. Logo \[ t-1 < x \ee x
    \leq t .\] Agora, suponhamos $t-1 < x \ee x \leq t .$ Então, $ x
    \leq t$ e $\lceil x \rceil = \min \{ z\in \Z \colon z\geq x \}$
    implica em $\lceil x \rceil \leq t$, e de $t-1 < x$ e
    \eqref{eq:teto} temos $t-1 \leq \lceil x \rceil $, ou seja, $t
    \leq \lceil x \rceil $ pois  $ t,\lceil x \rceil \in \Z$.
% Achei que a 2a. parte do parágrafo acima ficou meio nebulosa... a
% partir do "Agora, suponhamos..."...
%   Zeh

  \item Vamos provar 3. Suponhamos que $\lceil x \rceil = t$. Se
    $\lceil x \rceil = t$ então $t \geq x$. Do item 2 temos $t-1 < x$,
    ou seja, $t < x +1$. Portanto, \[ x \leq t \ee t < x+1.\] Agora,
    suponhamos que $x \leq t \ee t < x+1.$ De $t < x+1$ e $x+1 \leq
    \lceil x \rceil +1$ temos $t\leq \lceil x \rceil$. Por outro lado,
    se $x \leq t$ então $t\in \{ z\in \Z \colon z\geq x \}$, portanto
    $t\geq \lceil x \rceil$, logo $t =\lceil x \rceil $.

  \item Vamos provar 4, ou seja, vamos provar que para $x,y \in \R$
    \[ x < y \Rightarrow \lceil x \rceil \leq \lceil y \rceil.\] Se $x
    < y$ e $ y \leq \lceil y \rceil$ então $ x \leq \lceil y \rceil$,
    portanto $\lceil y \rceil \in \{ z\in \Z \colon z\geq x \}$,
    portanto, $\lceil x \rceil \leq \lceil y \rceil$.

  \item Finalmente, vamos provar 5.  Usando o item 3 temos 
    \[ x \leq \lceil x \rceil  < x+1\] portanto
    \[ x + t \leq \lceil x \rceil + t  < x+t+1\] e pelo item 3, novamente,
    \[ \lceil x \rceil + t  = \lceil x  + t \rceil.\]
  \end{itemize}
\end{proof}

\begin{teorema}\label{teo:35}
  Se $f\colon \R \to \R$ é contínua, crescente e tal que \[ f(x) \in
  \Z \Rightarrow x \in \Z\] então \[ \big\lceil f ( \lceil x \rceil )
  \big\rceil = \lceil f ( x ) \rceil .\]
\end{teorema}
\begin{proof}
  Seja $f\colon \R \to \R$ contínua, crescente e tal que $f(x) \in \Z
  \Rightarrow x \in \Z$, vamos provar que $\big\lceil f ( \lceil x
  \rceil ) \big\rceil = \lceil f ( x ) \rceil$ em dois casos. Sabemos
  por \eqref{eq:teto} que $x \leq \lceil x \rceil$.

  Se $x = \lceil x \rceil$ então $\big\lceil f ( \lceil x \rceil )
  \big\rceil = \lceil f ( x ) \rceil$. Resta provar o caso $x < \lceil
  x \rceil$.

  Seja $x$ um real tal que $x < \lceil x \rceil$. Se $f$ é crescente,
  então $f(x) < f \big(\lceil x \rceil\big)$.  Tomemos o intervalo $[
  f(x), f \big(\lceil x \rceil\big) ]$. Como $f$ é contínua, para todo
  $y \in [ f(x), f \big(\lceil x \rceil\big)]$ existe $a \in [x,
  \lceil x \rceil ]$ tal que $f(a)=y$. 

  Se $f(x) < f \big(\lceil x \rceil\big)$ então $\lceil f(x) \rceil
  \leq \lceil f \big(\lceil x \rceil\big)\rceil$ pelo item 4 do
  teorema \ref{teo:teto}. 

  Suponhamos que $\lceil f(x) \rceil < \lceil f \big( \lceil x
  \rceil \big) \rceil$.  Se $\lceil f(x) \rceil > f \big( \lceil x
  \rceil \big)$ então $\lceil f(x) \rceil$ é um inteiro entre $f(
  \lceil x \rceil )$ e $ \lceil f( \lceil x \rceil )\rceil$.

  Então $\lceil f(x) \rceil \in \big[ f(x), f \big(\lceil x
  \rceil\big)\big)$, portanto, existe $a \in [x, \lceil x \rceil)$ tal
  que $f(a)= \lceil f(x) \rceil$. Logo $f(a)\in \Z$ e, por hipótese,
  $a\in \Z$, portanto $a \not\in [x, \lceil x \rceil)$, uma
  contradição.

  Com isso temos $\lceil f(x) \rceil \leq \lceil f \big(\lceil x
  \rceil\big)\rceil$ e $\nao(\lceil f(x) \rceil < \lceil f \big(\lceil
  x \rceil\big)\rceil)$, ou seja, $\lceil f(x) \rceil = \lceil f
  \big(\lceil x \rceil\big)\rceil$.
\end{proof}

\begin{teorema}\label{teo:chao}
  Sejam $x \in \R$ e $t\in \Z$.\\
  $\lfloor \; \rfloor \colon \R \to \Z$ satisfaz
  \begin{enumerate}

  \item $\lfloor \; \rfloor$ é não-decrescente em $\R$,
  \item $\lfloor x+t \rfloor = \lfloor x \rfloor + t$.
  \item Se $f\colon \R \to \R$ é contínua, crescente e tal que \[ f(x)
    \in \Z \Rightarrow x \in \Z\] então \[ \big\lfloor f ( \lfloor x
    \rfloor ) \big\rfloor = \lfloor f ( x ) \rfloor .\]
  \end{enumerate}
\end{teorema}

\textbf{Exemplo.} \\
A função definida nos reais $f(x) = \frac xk$, com $k\in\Z$ é contínua
(pois é uma função linear) crescente e tal que $ f(x) \in \Z
\Rightarrow x \in \Z$ (pois $f(x) = t$ implica $x=kt$), portanto
\[ \left\lfloor \frac {\lfloor x/k \rfloor}{k} \right\rfloor =
 \left\lfloor \frac x{k^2} \right\rfloor .\] 



\begin{teorema}

\end{teorema}
\begin{proof}

  portanto
  \[-x \geq -\lceil x \rceil > -x-1\]

  \[  -\lceil x \rceil = \lfloor -x \rfloor .\]
  
  A prova de $\lceil - x \rceil =   - \lfloor x \rfloor$ é exercício.
\end{proof}

Seja $n_0\in\N$ e $A = \{n\in\N\colon n \geq n_0\}$. Podemos definir
uma função com domínio $A$ usando indução da seguinte maneira:
\begin{itemize}
\item (base) especificamos o valor de $f(n_0)$,
\item (passo) especificamos para todo $n \geq n_0$ como determinar $f(n+1)$
  a partir de todos ou alguns elementos de $\{f(n_0), f(n_0+1),\dots,
  f(n-1),f(n)\}.$
\end{itemize}
Se $X\subseteq A$ é o conjunto dos pontos para os quais $f$ está definida
então pelo Princípio da Indução, teorema \ref{teo:PIF3}, $X=A$.

\textbf{Exemplos}
\begin{description}
\item[(1)] $f(0)=3$ e $\paratodo n\geq 0,~f(n+1)=2f(n)+3$.
\item[(2)] $F(0)=1$ e $\paratodo n\geq 0,~F(n+1)=(n+1)F(n)$.
\item[(3)] $S(0)=0$ e $\paratodo n\geq 0,~S(n+1)=S(n)+g(n+1)$, em que
  $g$ é uma função
\item[(4)] $p(0)=1$ e $\paratodo n\geq 0, ~ p(n+1)=ap(n)$, em que $a\in

\item[(5)] $b(1)=1$ e $\paratodo n\geq 1, ~ b(n+1)=b\left( \left\lfloor
      \frac{n+1}2 \right\rfloor \right) +1$.
\item[(6)] $H(1)=1$ e $\paratodo n\geq 1, ~ H(n+1) = H (n) + \frac 1{n+1}$.
\end{description}

Notemos que nos exemplos acima a função $F$ é $F(n)=n!$, a função $S$
é $S(n) = \sum_{i=1}^n g(i)$ e a função $p$ é a função $p(n)= a^n$. O
que podemos dizer a respeito das outras funções? Vamos provar que

Se $n = 0$ então $f(0)=3$ e $g(0) = 3(2-1) =3$, portanto $f(0) =

Agora, vamos provar que para todo $n\in \N$, se $f(n)=g(n)$ então

  &=& 3(2^{n+2}-1)\\
  &=& g(n+1).
\end{eqnarray*}
a primeira igualdade é a definição de $f$, na segunda foi usada a
hipótese da indução, nas terceira e na última a definição de $g$. 

Portanto, pelo teorema \ref{teo:PIF1} $\paratodo n, ~f(n)=g(n)$.

  b\left( \left\lfloor \frac{n}{2^4} \right\rfloor \right) + 4  \\
  &\vdots& \cdots \\
  &=& b\left( \left\lfloor \frac{n}{2^i} \right\rfloor \right) + i 
\end{eqnarray*}
e pelo teorema \ref{teo:chao}
\begin{eqnarray*}
  \left\lfloor \frac n{2^i} \right\rfloor = 1 \Leftrightarrow 1 \leq \frac n{2^i} < 2 
  \Leftrightarrow i \leq \log_2 n < i+1 \Leftrightarrow i = \lfloor \log_2 n \rfloor
\end{eqnarray*}
portanto, após $\lfloor \log_2 n \rfloor$ iterações $b(n) = b(1) +
\lfloor \log_2 n \rfloor = 1+ \lfloor \log_2 n \rfloor$.
Resta provar por indução que $b(n) = 1+ \lfloor \log_2 n \rfloor$.

Nem sempre funciona, ou as vezes é muito difícil chegar em algum
lugar, por exemplo, tente o mesmo com \[ F(1)=F(2)=1 \ee F(n+1) = F(n)
+ F(n-1),~\mathrm{se}~ n>1.\] Por outro lado, em análise de algoritmos
muitas vezes é suficiente uma solução assintótica, isto é, basta saber
a ordem de grandeza da solução. Por exemplo, no caso da função $F$
definida acima, podemos provar que $F(n) = \Theta ( \varphi^n)$, onde
$\varphi = \frac{1+\sqrt{5}}2$ por indução em $n$.

De fato, defina $c=1/2$, $C=1$, e temos que verificar a base da
indução $n=1$ e $n=2$ (exercício) que nesses casos \[ \paratodo k\in \{1,2\}, ~c
\varphi^k \leq F(k) \leq C \varphi^k \]

Seja $n>1$ um natural e assuma que \[ \paratodo k\in [1..n], ~c
\varphi^k \leq F(k) \leq C \varphi^k \] Então
\begin{eqnarray*}
  F(n+1) &=& F(n) + F(n-1)\\
  &\geq& c \varphi^n +  c \varphi^{n-1} 
  =  c \varphi^{n-1}(\varphi  +  1) 
  =  c \varphi^{n-1}(\varphi^2)  
  =  c \varphi^{n+1} 
\end{eqnarray*}e 
\begin{eqnarray*}
  F(n+1) &=& F(n) + F(n-1)\\
  &\leq& C \varphi^n +  C\varphi^{n-1} 
  =  C \varphi^{n-1}(\varphi  +  1) 
  =  C \varphi^{n-1}(\varphi^2)  
  =  C \varphi^{n+1} 
\end{eqnarray*}

Considere a seguinte função definida indutivamente (recorrência)
\begin{eqnarray*}
   T(1) &=& 0,\\ 
   T(2) &=& 1,\\ 
   T(n) &=& T \left(\left\lceil \frac n2 \right\rceil \right) +
   T \left(\left\lfloor \frac n2 \right\rfloor \right) + 2n -1  
   ~\textrm{ se } ~n>2.
\end{eqnarray*}
Vamos provar a seguinte solução assintótica $T(n) = O(n^2)$. Para tal,
vamos provar por indução em $n$ que existe $c,n_0>0$ tais que \[ T(n)
\leq c n^2 ~\paratodo n\geq n_0.\]

Tomemos $c=1$ e $n_0=6$. 

Para $n\in [2..6]$ temos 
\begin{eqnarray*}
  T(1)  =  0 ~ &\textrm{ e }&~ c{1}^2 = 1\\
  T(2)  =  2 ~ &\textrm{ e }&~ c{2}^2 = 4\\
  T(3)  =  6 ~ &\textrm{ e }&~ c{3}^2 = 9\\
  T(4)  =  11~ &\textrm{ e }&~ c{4}^2 = 16\\
  T(5)  =  17~ &\textrm{ e }&~ c{5}^2 = 25\\
  T(6)  =  25~ &\textrm{ e }&~ c{6}^2 = 36\\
\end{eqnarray*}
portanto, $T(n) \leq c n^2 ~\paratodo n\in[2..6]$. Esses são o caso
base. Seja $n>6$ um inteiro e assuma que para todo $k\in [n_0..n-1]$,
$T(k) \leq c k^2$. Vamos provar que $T(n) \leq c n^2$. Por definição
$T(n) = T \left(\left\lceil \frac n2 \right\rceil \right) + T
\left(\left\lfloor \frac n2 \right\rfloor \right) + 2n -1$ e usando a
hipótese da indução
\begin{eqnarray*}
  T(n) & \leq & c \left\lceil \frac n2 \right\rceil^2 + 
  c \left\lfloor \frac n2 \right\rfloor^2 + 2n -1 \\
  & \leq & c \left( \frac n2 +1 \right)^2 + 
  c \left( \frac n2 \right)^2 + 2n -1\\
  & =& \frac c2 n^2 + (c+2)n
\end{eqnarray*}
portanto, $T(n) \leq cn^2$ para todo $n\geq 6$.

Essa solução está superestimada, de fato é possível provar uma
\begin{red}
  solução justa
\end{red}
ou seja, uma solução que limita assintoticamente superiormente e
inferiormente a função $T(n)$.

Vamos provar que $T(n) = \Theta(n\log_2 n)$.

Primeiro vamos provar que $T(n) = O ( n \lceil \log_2(n)
\rceil)$. Para tal, tomemos $c=2$ e $n_0=1$ e vamos provar que \[ T(n)
\leq c  n \lceil \log_2(n)\rceil ~\paratodo n\geq n_0.\]

Para $n=1$ temos $T(1)=0$ e $c 1 \lceil \log_2(1)\rceil =0$ e
para $n=2$ temos $T(2)=1$ e $c 2 \lceil \log_2(2)\rceil =2$,
portanto, $T(n) \leq c n \lceil \log_2(n)\rceil$ nesses casos.

Seja $n>2$ inteiro e assumamos que \[ T(k) \leq c k \lceil
\log_2(k)\rceil ~\paratodo k\in[n_0..n-1].\] Por definição \[ T(n) = T
\left(\left\lceil \frac n2 \right\rceil \right) + T \left(\left\lfloor
    \frac n2 \right\rfloor \right) + 2n -1 \]
e por hipótese
\[ T(n) \leq c \left\lceil \frac n2 \right\rceil \left\lceil \log_2
  \left( \left\lceil \frac n2 \right\rceil \right)\right\rceil + c
\left\lfloor \frac n2 \right\rfloor \left\lceil \log_2 \left(
    \left\lfloor \frac n2 \right\lfloor \right)\right\rceil +2n -1
\]
ainda, pelo teorema \ref{teo:35}
\[ \left\lceil \log_2
  \left( \left\lceil \frac n2 \right\rceil \right)\right\rceil = 
\left\lceil \log_2
  \left( \frac n2  \right)\right\rceil
\]
e como $\lfloor x \rfloor \leq x$ e $\log_2$ é crescente
\[ 
\left\lceil \log_2 \left(
    \left\lfloor \frac n2 \right\lfloor \right)\right\rceil \leq
\left\lceil \log_2 \left(
    \frac n2  \right)\right\rceil
\]
portanto
\begin{eqnarray*}
  T(n) &\leq& c \left\lceil \frac n2 \right\rceil \left\lceil \log_2
    \left(   \frac n2 \right)\right\rceil + c
  \left\lfloor \frac n2 \right\rfloor \left\lceil \log_2 \left(
      \frac n2  \right)\right\rceil +2n -1\\
  & = &  c \left( \left\lceil \frac n2 \right\rceil + 
    \left\lfloor \frac n2 \right\rfloor  \right) \left\lceil \log_2 \left(
      \frac n2  \right)\right\rceil +2n -1\\
  & = &  c  n \left\lceil \log_2 \left(
      (n) -1 \right)\right\rceil +2n -1\\
  & = &  c  n \left\lceil \log_2   n  \right\rceil -(c-2) n -1\\
  & \leq & c  n \left\lceil \log_2  n\right\rceil .
\end{eqnarray*}

Com isso provamos que $T(n) = O ( n \lceil \log_2(n) \rceil)$. Ainda
temos que $n= O(n)$ e $\lceil \log_2(n) \rceil = O( \log_2 (n) )$,
portanto, pelo teorema \ref{teo:O4}, $n \lceil \log_2(n) \rceil =
O(n\log_2 (n) )$, e pelo teorema \ref{teo:O3} temos, finalmente, que
$T(n) = O ( n \log_2(n))$.

Agora, vamos provar que $T(n) = \Omega ( n \log_2(n) )$ e com isso
teremos que $T(n) = \Theta(n\log_2 n)$. Para tal, tomemos $c=1/2$ e
$n_0=1$ e vamos provar que \[ T(n) \geq c n \lfloor \log_2(n)\rfloor
~\paratodo n\geq n_0.\]

Para $n=1$ temos $T(1)=0$ e $c 1 \lfloor \log_2(1)\rfloor =0$ e para
$n=2$ temos $T(2)=1$ e $c 2 \lfloor \log_2(2)\rfloor = 1$, portanto,
$T(n) \geq c n \lfloor \log_2(n)\rfloor$ nesses casos.

Seja $n>2$ inteiro e assumamos que \[ T(k) \geq c k \lfloor
\log_2(k)\rfloor ~\paratodo k\in[n_0..n-1].\] Por definição \[ T(n) = T
\left(\left\lceil \frac n2 \right\rceil \right) + T \left(\left\lfloor
    \frac n2 \right\rfloor \right) + 2n -1 \]
e por hipótese
\[ T(n) \geq c \left\lceil \frac n2 \right\rceil \left\lfloor \log_2
  \left( \left\lceil \frac n2 \right\rceil \right)\right\rfloor + c
\left\lfloor \frac n2 \right\rfloor \left\lfloor \log_2 \left(
    \left\lfloor \frac n2 \right\rfloor \right)\right\rfloor +2n -1
\]
ainda, pelo teorema \ref{teo:chao}
\[ \left\lfloor \log_2 \left( \left\lfloor \frac n2 \right\rfloor
  \right)\right\rfloor = \left\lfloor \log_2 \left( \frac n2
  \right)\right\rfloor
\]
e como $\lceil  x \rceil \geq x$ e $\log_2$ é crescente
\[ 
\left\lfloor \log_2 \left(
    \left\lceil \frac n2 \right\lceil  \right)\right\rfloor \geq
\left\lfloor \log_2 \left(
    \frac n2  \right)\right\rfloor
\]
portanto
\begin{eqnarray*}
  T(n) &\geq& c \left\lceil \frac n2 \right\rceil \left\lfloor \log_2
    \left(   \frac n2 \right)\right\rfloor + c
  \left\lfloor \frac n2 \right\rfloor \left\lfloor \log_2 \left(
      \frac n2  \right)\right\rfloor +2n -1\\
  & = &  c \left( \left\lceil \frac n2 \right\rceil + 
    \left\lfloor \frac n2 \right\rfloor  \right) \left\lfloor \log_2 \left(
      \frac n2  \right)\right\rfloor +2n -1\\
  & = &  c  n \left\lfloor \log_2 
      (n) -1  \right\rfloor +2n -1\\
  & = &  c  n \left\lfloor \log_2   n  \right\rfloor -(c-2) n -1\\
  & \geq & c  n \left\lfloor \log_2  n\right\rfloor .
\end{eqnarray*}

Com isso provamos que $T(n) = \Omega ( n \lfloor \log_2(n)
\rfloor)$. Ainda temos que $n= \Omega (n)$ e $\lfloor \log_2(n)
\rfloor = O( \log_2 (n) )$ (prove), portanto, $n \lfloor \log_2(n)
\rfloor = \Omega (n\log_2 (n) )$ (prove), finalmente, temos que $T(n)
= \Omega ( n \log_2(n))$ (prove).

Notemos que de $T(n) = \Omega ( n \log_2(n))$ não vale que $T(n) =
O(n)$. Onde está o erro da seguinte prova.

Vamos provar que existem $c,n_0>0$ tais que $T(n) \leq cn$ para todo
$n\geq n_0$. Tomemos $c=1/2$ e $n_0=1$.

Para $n=1,2$ a desigualdade vale (verifique). Seja $n>2$ inteiro e
assumamos que para todo $k\in [n_0..n-1]$, $T(k) \leq c k$. Vamos
mostrar que $T(n) \leq cn$.
\begin{eqnarray*}
  T(n) &=& T \left(\left\lceil \frac n2 \right\rceil \right) +
   T \left(\left\lfloor \frac n2 \right\rfloor \right) + 2n -1  \\
   & \leq & c \left\lceil \frac n2 \right\rceil +
            c \left\lfloor \frac n2 \right\rfloor + 2n -1\\
       &=& c n + 2n -1\\
       &=& (c+2) n -1
\end{eqnarray*}
e como $c$ é constante, também $c+2$ é constante, portanto $T(n) = O(n)$.

Vamos usar a mesma técnica para mostrar que $b(1)=1$, $b(n+1)=
b(\lfloor (n+1)/2 \rfloor ) +1$ tem solução assintótica $b(n) = \Theta

Primeiro, vamos mostrar que $b(n) = O(\lfloor \log_2 n \rfloor)$.

Vamos mostrar que para $c=3$ e $n_0 = 2$, $b(n) \leq c \lfloor \log_2
n \rfloor$ para todo $n \geq n_0$.

Para $n=2,3$ temos $b(2) = b(3) = 2$ e $ c \lfloor \log_2 2 \rfloor =

[n_0..n-1]$, $b(k) \leq c \lfloor \log_2 k \rfloor$. Então
\begin{eqnarray*}
  b(n) &=& b(\lfloor n/2 \rfloor) +1 \\
  & \leq & c \lfloor \log_2 \lfloor n/2 \rfloor \rfloor +1\\
  & = & c \lfloor \log_2 ( n/2)  \rfloor +1\\
  & = & c \lfloor \log_2(n) -\log_2(2) \rfloor +1\\
  & = & c \lfloor \log_2(n) \rfloor -1 +1\\
  & = & c \lfloor \log_2(n) \rfloor 
\end{eqnarray*}
portanto $b(n) \leq c \lfloor \log_2 n \rfloor$ para todo $n \geq

$\lfloor \log_2 n \rfloor = O(\log_2 n )$, usando o teorema 41 temos
$b(n) = O(\log_2 n )$.

Agora, vamos mostrar que $b(n) = \Omega (\lfloor \log_2 n \rfloor)$.

Vamos mostrar que para $c=1$ e $n_0 = 2$, $b(n) \geq c \lfloor \log_2

Para $n=2,3$ temos $b(2) = b(3) = 2$ e $ c \lfloor \log_2 2 \rfloor =

  b(n) &=& b(\lfloor n/2 \rfloor) +1 \\
  & \geq & c \lfloor \log_2 \lfloor n/2 \rfloor \rfloor +1\\
  & = & c \lfloor \log_2 ( n/2)  \rfloor +1\\
  & = & c \lfloor \log_2(n) -\log_2(2) \rfloor +1\\
  & = & c \lfloor \log_2(n) \rfloor -1 +1\\
  & = & c \lfloor \log_2(n) \rfloor 
\end{eqnarray*}
portanto $b(n) \geq c \lfloor \log_2 n \rfloor$ para todo $n \geq

Vimos que $f$ dada por $f(0)=3$ e $f(n+1)=2f(n)+3$, se $n\geq 0$, é

Tomemos $c=3$ e $n_0=0$.

Para $n=0$ temos $f(0) = 3$ e $ c 2^{n+1} - 3 = 3$, portanto $f(0) \

\end{eqnarray*}
Logo $f(n) = O(2^{n+1} - 3 )$ e como $2^{n+1} - 3 = O(2^{n+1})$ temos
